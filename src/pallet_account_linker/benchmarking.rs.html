const SEED: u32 = 0;

benchmarks! {
	link_eth {
		let caller = account("caller", 0, 0);
		let account_id: T::AccountId = account("Alice", 0, SEED);

	link_polkadot {
		let caller = account("caller", 0, 0);
		let linked_account: T::AccountId = account("Alice", 0, SEED);
		let index: u32 = 0;
